deaths %>%

  ggplot(aes(x = age,

                      scale = 0.9)  +
  theme_ridges(font_size = 10) +
  scale_x_continuous(breaks = seq(0, 100, 10),
                     labels = seq(0, 100, 10)) +

  ylab("Year") +
